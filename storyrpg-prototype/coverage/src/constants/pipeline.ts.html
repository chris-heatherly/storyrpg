function dominantAttribute(def: SkillDefinition): string {
  let best = '';
  let bestWeight = 0;
  for (const [attr, w] of Object.entries(def.attributeWeights)) {
    if ((w ?? 0) > bestWeight) { bestWeight = w ?? 0; best = attr; }
  }
  return best;
}
